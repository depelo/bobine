function authenticateToken(req, res, next) {
    const token = req.cookies && req.cookies.jwt_token;
    if (!token) {
        return res.status(401).send('Token mancante');
    }
    jwt.verify(token, JWT_SECRET, (err, user) => {
        if (err) {
            return res.status(403).send('Token non valido');
        }
        req.user = user;
        next();
    });
}

const payload = {

Endpoint 1: GET /api/admin/users

Query su [CMP].[dbo].[Users] per restituire l’anagrafica globale (Passaporti):

TargetTable AS targetTable



RoleDefinition AS roleDefinition



Dopo il fetch, mappare result.recordset per:

Applicare JSON.parse al campo roleDefinition così che il frontend riceva un oggetto JS e non una stringa JSON.



Gestire eventuali errori di database con 500 come sopra.



(Opzionale) Se si vuole maggiore robustezza, racchiudere il JSON.parse in un try/catch interno che, in caso di errore di parsing, logga e rimanda il valore originale o null, ma per ora attenersi alla specifica fornita.



Coerenza con CORS e cookie





L’app già usa cookieParser() e serve i file statici dalla stessa origin (express.static(__dirname)), quindi le chiamate da captain.html a /api/... useranno i cookie jwt_token in modo implicito se credentials: 'include' è impostato a livello di fetch.



Non sono necessarie modifiche a CORS in quanto le richieste arrivano dalla stessa origin.
